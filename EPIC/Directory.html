def each
  dirname = "./public#{self.path.unescape}"
  Dir.open dirname do
    |dir|
    dir.each do
      |filename|
      filepath = dirname + filename
      yield Rackful::Path.new( self.path + escape_path(filename) ) \
        if File::file? filepath
    end
  end
end
