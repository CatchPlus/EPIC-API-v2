def each
  ActiveHandleValue.select(:handle).uniq.
    where('`handle` LIKE ?', self.prefix + '/%').
    find_each do |ahv|
      suffix = %r{\A[^/]+/(.*)}.match(ahv.handle)[1]
      yield( { :uri => suffix.escape_path, :name => "#{prefix}/#{suffix}" } )
    end
end
