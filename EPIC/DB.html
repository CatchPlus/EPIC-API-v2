def each_handle prefix = nil
  ds = self.pool[:handles].select(:handle).distinct
  if prefix
    ds = ds.filter( '`handle` LIKE ?', prefix + '/%' )
  end
  self.sql_depth = self.sql_depth + 1
  begin
    ds.each { |row| yield row[:handle] }
  ensure
    self.sql_depth = self.sql_depth - 1
  end
end
