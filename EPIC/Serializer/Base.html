def recurse?
  depth = ( self.resource.class.constants.include? :DEFAULT_DEPTH ) ?
    self.resource.class::DEFAULT_DEPTH.to_s : '0'
  depth = request.env['HTTP_DEPTH'] || depth
  requested? && '0' != depth || 'infinity' == depth
end
