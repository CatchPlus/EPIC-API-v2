def each
  yield '{'
  first = true
  self.resource.each do
    |item|
    uri = item[:uri]
    #if 0 == uri.index(self.resource.path)
    #  uri = uri[Range.new(self.resource.path.length, -1)]
    #end
    item = item.dup
    item.delete :uri
    yield( (first && '' || ',') + uri.to_json + ':' + ::JSON.generate(item) )
    first = false
  end
  yield '}'
end
